from hypothesis import given
from hypothesis.strategies import integers

class TestFib(unittest.TestCase):
    @given(k=integers(min_value=2))
    def test_fib(self, k):
        self.assertTrue(fib(k), fib(k-1) + fib(k-2))
\end{minted}
