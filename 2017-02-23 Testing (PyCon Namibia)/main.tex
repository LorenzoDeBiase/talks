import unittest

def fib(n):
    """Returns the n th fibonacci number.
    For example:

        >>> fib(5)
        5
        >>> fib(6)
        8
   """
    if n == 1:
        return 1
    elif n == 0:
        return 0
    else:
        return fib(n-1) + fib(n-2)
\end{minted}
