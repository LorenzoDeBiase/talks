    {\frac{1}{2} p \bar{Q} p^T + \bar{c}^T p + \bar{a}}\]

    \begin{itemize}
      \item \boldmath\(Q, \bar{Q} \in\mathbb{R}^{4 \times 4}\)
      \item \boldmath\(c, \bar{c}\in\mathbb{R}^{4 \times 1}\) 
      \item \boldmath\(a, \bar{a}\in\mathbb{R}\)  
   \end{itemize}
    \end{lemma}
    \end{center}
